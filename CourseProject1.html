data$DateTime <- as.POSIXct(paste(data$Date, " ", data$Time), format = "%d/%m/%Y %H:%M:%S")
data$Date <- as.Date(data$Date, format = "%d/%m/%Y")
data$Time <- strptime(data$Time, format = "%H:%M:%S")
data = subset(data, Date >= "2007-02-01" & Date <= "2007-02-02")
data$Global_active_power = as.numeric(type.convert(as.character(data$Global_active_power), 
    dec = "."))
data$Global_reactive_power = as.numeric(type.convert(as.character(data$Global_reactive_power), 
    dec = "."))
data$Voltage = as.numeric(type.convert(as.character(data$Voltage), dec = "."))
data$Global_intensity = as.numeric(type.convert(as.character(data$Global_intensity), 
    dec = "."))
data$Sub_metering_1 = as.numeric(type.convert(as.character(data$Sub_metering_1), 
    dec = "."))
data$Sub_metering_2 = as.numeric(type.convert(as.character(data$Sub_metering_2), 
    dec = "."))
data$Sub_metering_3 = as.numeric(type.convert(as.character(data$Sub_metering_3), 
    dec = "."))

Plot 1

hist(data$Global_active_power, main = "Global active power", xlab = "Global active power (kilowatts)", 
    ylab = "Frequency", col = "red")

Plot 2

plot(data$DateTime, data$Global_active_power, type = "l", ylab = "Global active power (kilowatts)", 
    xlab = "")

Plot 3

plot(range(data$DateTime), range(cbind(data$Sub_metering_1, data$Sub_metering_2, 
    data$Sub_metering_3)), ylab = "Energy sub metering", xlab = "", type = "n")
lines(data$DateTime, data$Sub_metering_1)
